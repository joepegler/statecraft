    ```ts [index.ts]
    import { test, expect } from "vitest";
    import { erc20Abi, parseEther } from "viem";
    import {
      scenario,
      withChain,
      withDeployments,
      withErc20Balance,
      withFundedWallet,
      withBundler,
    } from "@statecraft/vitest";

    import { answerArtifact, USDC } from "./config.js";

    test(
      "funded wallet has ETH, USDC, and a deployed contract",
      scenario(
        withChain(),
        withFundedWallet({ balance: parseEther("1") }),
        withDeployments({
          answer: {
            artifact: answerArtifact,
            args: [],
          },
        }),
        withErc20Balance({
          token: USDC,
          amount: 1_000_000n,
        }),
        withBundler(),
        async ({ publicClient, walletClient, deployments, bundlerClient }) => {
          const ethBalance = await publicClient.getBalance({ address: walletClient.address });
          // Smoke test: proves fixture wiring.
          expect(ethBalance).toEqual(parseEther("1"));
        },
      ),
    );
    ```
